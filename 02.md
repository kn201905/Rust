* 配列
```rust
use std::mem;

// スライスの借用
fn analyze_slice(slice: &[i32]) {
	println!("1st: {}", slice[0]);
	println!("len: {}", slice.len());
}

fn main() {
	let xs: [i32; 5] = [1, 2, 3, 4, 5];
	
	// すべての要素を0にする場合
	let ys: [i32; 500] = [0; 500];

	// Arrays are stack allocated
	// 配列はスタック上に置かれる
	println!("xs occupies {} bytes", mem::size_of_val(&xs));
	println!("ys occupies {} bytes", mem::size_of_val(&ys));
	
	// 配列は自動的にスライスとして借用される。
	analyze_slice(&xs);
	analyze_slice(&ys[1 .. 4]);

	// インデックスの範囲が配列のサイズを超えた場合パニックする
	// println!("{}", xs[5]);
}
```

# 構造体

```rust
#[derive(Debug)]
struct Person<'a> {
	name: &'a str,
	age: u8,
}

// A unit struct
struct Nil;

//「名前付き」タプル
struct Pair(i32, f32);

// 構造体
#[derive(Debug)]
struct Point {
	x: f32,
	y: f32,
}

// 構造体は他の構造体のフィールドになることができる
#[allow(dead_code)]
#[derive(Debug)]
struct Rectangle {
	top_left: Point,
	bottom_right: Point,
}

fn main() {
	let peter = Person { name: "Peter", age: 27 };
	println!("\n{:?}", peter);

	let point: Point = Point { x: 10.3, y: 0.4 };
	println!("\npoint: ({}, {})", point.x, point.y);

	let bottom_right = Point { x: 5.2, ..point };
	println!("bottom_right: ({}, {})", bottom_right.x, bottom_right.y);

	// Destructure the point using a `let` binding
	let Point { x: top_edge, y: left_edge } = point;

	let _rectangle = Rectangle {
		top_left: Point { x: left_edge, y: top_edge },
		bottom_right: bottom_right,
	};
	println!("_rectangle: {:?}", _rectangle);

	// Instantiate a unit struct
	let _nil = Nil;

	//「名前付き」タプル
	let pair = Pair(1, 0.1);

	//「名前付き」タプルの分解
	let Pair(integer, decimal) = pair;
	println!("\npair contains {:?} and {:?}", integer, decimal);
}
```

```rust
enum WebEvent {

	PageUnload,
	KeyPress(char),

	match event {
		WebEvent::PageLoad => println!("page loaded"),
		WebEvent::PageUnload => println!("page unloaded"),

		WebEvent::Paste(s) => println!("pasted \"{}\".", s),

	let pressed = WebEvent::KeyPress('x');
	// `to_owned()` creates an owned `String` from a string slice.

	let load    = WebEvent::PageLoad;
	let unload  = WebEvent::PageUnload;

	inspect(pressed);
	inspect(pasted);
	inspect(click);
	inspect(load);
	inspect(unload);
}
```

* 列挙型の例（linked list）
```rust
// 以下の３通りのどの書き方でもＯＫ
use self::List::*;
//use crate::List::*;
//use self::List::*;

enum List {
	// Cons と Nil は、単なるシグネチャ（整数）
	Cons(u32, Box<List>),
	Nil,
}

// 列挙型にはメソッドを付与することができる。
impl List {
	fn new() -> List { Nil }
	fn prepend(self, val: u32) -> List { Cons(val, Box::new(self)) }
	
	fn len(&self) -> u32 {
		// `self`の型は`&List`であるので、`*self`は`List`になる。
		// match は、リファレンス(`&T`)ではなく実体(`T`)に対して行うのが好ましい。
		match *self {
			// `self`をすでに借用しているので、tailの所有権を取ることができない。
			// 代わりに参照を使用する。
			Cons(_, ref tail) => 1 + tail.len(),
			Nil => 0
		}
	}

	// Listをheap上の文字列として表したものを返すメソッド。
	fn stringify(&self) -> String {
		match *self {
			Cons(head, ref tail) => {
				// `format!`は`print!`に似ているが、コンソール上に出力
				// する代わりに、heap上の文字列を返す。
				format!("{}, {}", head, tail.stringify())
			},
			Nil => { format!("Nil") },
		}
	}
}

fn main() {
	let mut list = List::new();
	
	list = list.prepend(1);
	list = list.prepend(2);
	list = list.prepend(3);

	println!("len: {}", list.len());
	println!("list: {}", list.stringify());
}
```

# エイリアス

```rust
enum VeryVerboseEnumOfThingsToDoWithNumbers {
	Add,
	Subtract,
}

// Self エイリアス
impl VeryVerboseEnumOfThingsToDoWithNumbers {
	fn run(&self, x: i32, y: i32) -> i32 {
		match self {
			Self::Add => x + y,
			Self::Subtract => x - y,
		}
	}
}	

// type エイリアス
type Operations = VeryVerboseEnumOfThingsToDoWithNumbers;

fn main() {
	// We can refer to each variant via its alias, not its long and inconvenient
	// name.
	let op_add = Operations::Add;
	let add = op_add.run(5, 3);
	println!("add: {}", add);
	
	println!("sub: {}", Operations::Subtract.run(5, 3));
}
```

# 値のフリーズ

```rust
fn main() {
    let mut _mutable_integer = 7i32;
    println!("_mutable_integer: {}", _mutable_integer);

    {
        // _mutable_integer の imuutable でのシャドーイング
        let _mutable_integer = _mutable_integer;

        // `_mutable_integer`はこのスコープではフリーズしているため、以下の行はエラーとなる
        // _mutable_integer = 50;
    }

    {
        // 以下は可能
        let mut _mutable_integer = _mutable_integer;
        _mutable_integer = 50;
        println!("_mutable_integer: {}", _mutable_integer);
    }
    println!("_mutable_integer: {}", _mutable_integer);
}
```
